CREATE TABLE User(
	UserName VARCHAR(15) NOT NULL,
	Fname VARCHAR(15) NOT NULL,
	Lname VARCHAR(30) NOT NULL,
	Role INT NOT NULL,
	Password VARCHAR(30) NOT NULL,
	PRIMARY KEY(UserName),
	FOREIGN KEY (Role)
		REFERENCES Roles(RoleD)
			ON DELETE SET DEFAULT
			ON UPDATE CASCADE
	);

CREATE TABLE Role(
	RoleID INT NOT NULL,
	RoleName VARCHAR(30) NOT NULL,
	PRIMARY KEY (RoleID)
	);
